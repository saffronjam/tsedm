// Dollars
(\$\d+)

// Websites
(https?:\/\/.*)

// Email
([a-zA-Z0-9\._-]{1,64})@([a-zA-Z.]{1,255})

// Phone numbers
(\d-\d{3}-\d{3}-\d{4})
(\d{3}-\d{3}-\d{4})
(\+\d+)

// Twnety-four seven
\d+\/(\d+)

// Time
(\d{2}:\d{2}:\d{2})

// Abbreviations, for example p.m.
([a-z]\.){2}

// IP
(\d+[\.]\d+[\.]\d+[\.]\d+)
(\d[\.\,]\d{1,3}[\.\,]\d{1,3})

// Shothand words
('em)
(\w+?[-']\w+)


// Snake case
((\w_?)+\.\w+)
